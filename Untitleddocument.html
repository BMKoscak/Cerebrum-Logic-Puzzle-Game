    <script>

        // --- Game Logic ---

        const MAX_MISTAKES = 5;

        const CHECKPOINT_INTERVAL = 10; // Save every 10 levels

        class Game {

            constructor() {

                this.level = 1;

                this.mistakes = 0;

                this.hints = 3;

                this.checkpoint = 1;

                this.currentPuzzle = null;

               

                // DOM Elements

                this.dom = {

                    level: document.getElementById('level-display'),

                    hints: document.getElementById('hints-display'),

                    integrityBar: document.getElementById('integrity-bar'),

                    integrityText: document.getElementById('integrity-text'),

                    puzzle: document.getElementById('puzzle-container'),

                    input: document.getElementById('answer-input'),

                    hintBox: document.getElementById('hint-box'),

                    hintText: document.getElementById('hint-text'),

                    hintBtn: document.getElementById('hint-btn'),

                    checkpoint: document.getElementById('checkpoint-display'),

                    lives: document.getElementById('lives-display'),

                    overlay: document.getElementById('overlay'),

                    overlayTitle: document.getElementById('overlay-title'),

                    overlayMsg: document.getElementById('overlay-message'),

                    overlayBtn: document.getElementById('overlay-btn'),

                    overlayIcon: document.getElementById('overlay-icon')

                };

                this.loadProgress();

                this.init();

            }

            init() {

                this.renderStats();

                this.generatePuzzle();

               

                // Enter key support

                this.dom.input.addEventListener('keypress', (e) => {

                    if (e.key === 'Enter') this.submitAnswer();

                });

                this.dom.overlayBtn.addEventListener('click', () => {

                    this.dom.overlay.classList.add('hidden');

                    this.dom.overlay.classList.remove('flex');

                    this.dom.input.focus();

                   

                    // If resetting from game over

                    if (this.mistakes >= MAX_MISTAKES) {

                        this.resetToCheckpoint();

                    }

                });

            }

            // --- Level Generation Engine ---

            generatePuzzle() {

                this.dom.input.value = '';

                this.dom.hintBox.classList.add('hidden');

                this.dom.input.focus();

               

                const lvl = this.level;

                let q = "", a = "", h = "";

                // Difficulty scaler

                if (lvl <= 5) {

                    // Simple Arithmetic Patterns

                    const start = Math.floor(Math.random() * 10) + 1;

                    const step = Math.floor(Math.random() * 5) + 2;

                    q = `${start}, ${start+step}, ${start+(step*2)}, ?`;

                    a = (start+(step*3)).toString();

                    h = `The numbers increase by ${step}.`;

                }

                else if (lvl <= 15) {

                    // Geometric / Multiplication

                    const start = Math.floor(Math.random() * 3) + 1;

                    const mult = 2;

                    q = `${start}, ${start*mult}, ${start*mult*mult}, ${start*mult*mult*mult}, ?`;

                    a = (start*Math.pow(mult, 4)).toString();

                    h = "Each number is multiplied by the previous factor.";

                }

                else if (lvl <= 25) {

                    // Fibonacci Variations

                    const f1 = Math.floor(Math.random() * 5);

                    const f2 = f1 + Math.floor(Math.random() * 3) + 1;

                    q = `${f1}, ${f2}, ${f1+f2}, ${f1+f2+f2}, ?`;

                    a = (f1+f2+f2 + f1+f2).toString();

                    h = "Add the previous two numbers together.";

                }

                else if (lvl <= 40) {

                    // Squares and Cubes

                    const start = Math.floor(Math.random() * 3) + 2;

                    if (Math.random() > 0.5) {

                        q = `2:4, 3:9, 4:16, ${start}:?`;

                        a = (start*start).toString();

                        h = "Number squared (n^2).";

                    } else {

                        q = `1, 8, 27, 64, ?`;

                        a = "125";

                        h = "Perfect cubes.";

                    }

                }

                else if (lvl <= 60) {

                    // Binary & Hex

                    const num = Math.floor(Math.random() * 15) + 10; // 10-25

                    if (Math.random() > 0.5) {

                        q = `BINARY: ${num.toString(2)} = ? (Decimal)`;

                        a = num.toString();

                        h = "Convert from base 2 to base 10.";

                    } else {

                        q = `DECIMAL: ${num} = ? (Hex)`;

                        a = num.toString(16).toUpperCase();

                        h = "Convert to base 16 (0-9, A-F).";

                    }

                }

                else if (lvl <= 80) {

                    // Interleaved Series (Two patterns mixed)

                    // e.g. +2, *2, +2, *2

                    const start = 5;

                    q = "5, 7, 14, 16, 32, 34, ?";

                    a = "68"; // *2

                    // Note: proceduralizing interleaved is complex, using templates for high levels

                    // Or simpler:

                    const n = Math.floor(Math.random() * 5) + 10;

                    q = `${n}, ${n-2}, ${n+2}, ${n}, ${n+4}, ?`;

                    a = (n+2).toString();

                    h = "Two alternating patterns or operations.";

                }

                else {

                    // Logic / Boss Levels

                    const riddles = [

                        {q: "1=3, 3=5, 5=4, 4=4. 6=?", a: "3", h: "Count the letters in the English word."},

                        {q: "8809=6, 7111=0, 2172=0, 6666=4, 1111=0, 3213=0, 7662=2, 9312=?", a: "1", h: "Count the closed circles in the digits."},

                        {q: "O, T, T, F, F, S, S, ?", a: "e", h: "First letter of numbers (One, Two...)"},

                        {q: "M, T, W, T, F, ?", a: "s", h: "Days of the week."},

                        {q: "J, F, M, A, M, J, ?", a: "j", h: "Months of the year."}

                    ];

                    // Cycle through riddles based on level mod

                    const r = riddles[lvl % riddles.length];

                    q = r.q;

                    a = r.a;

                    h = r.h;

                }

                this.currentPuzzle = { q, a, h };

                this.dom.puzzle.innerText = q;

            }

            // --- Core Mechanics ---

            submitAnswer() {

                let input = this.dom.input.value.trim().toLowerCase();

                let correct = this.currentPuzzle.a.toLowerCase();

                if (input === "") return;

                if (input === correct) {

                    this.handleSuccess();

                } else {

                    this.handleFailure();

                }

            }

            handleSuccess() {

                // Animation

                this.dom.input.classList.add('success-pulse');

                setTimeout(() => this.dom.input.classList.remove('success-pulse'), 500);

                // Update State

                this.level++;

                if (this.level % 5 === 0) this.hints++; // Gain hint every 5 levels

                if (this.level % CHECKPOINT_INTERVAL === 0 || this.level === 1) {

                    this.checkpoint = this.level;

                    this.showNotification("CHECKPOINT REACHED", `Progress saved at Level ${this.level}`, "success");

                    this.saveProgress();

                } else {

                    this.saveProgress(); // Auto save regular progress too

                }

                this.renderStats();

                this.generatePuzzle();

            }

            handleFailure() {

                this.mistakes++;

                this.dom.input.classList.add('shake');

                this.dom.input.value = '';

                setTimeout(() => this.dom.input.classList.remove('shake'), 500);

                this.renderStats();

                if (this.mistakes >= MAX_MISTAKES) {

                    this.triggerGameOver();

                }

            }

            useHint() {

                if (this.hints > 0) {

                    this.hints--;

                    this.dom.hintText.innerText = this.currentPuzzle.h;

                    this.dom.hintBox.classList.remove('hidden');

                    this.renderStats();

                } else {

                    // Shake the hint button if no hints

                    this.dom.hintBtn.classList.add('shake');

                    setTimeout(() => this.dom.hintBtn.classList.remove('shake'), 500);

                }

            }

            triggerGameOver() {

                this.showNotification(

                    "SYSTEM FAILURE",

                    `Too many errors. Reverting to Level ${this.checkpoint}.`,

                    "error"

                );

            }

            resetToCheckpoint() {

                this.mistakes = 0;

                this.level = this.checkpoint;

                this.hints = 3; // Reset hints or keep them? Let's reset to 3 for balance.

                this.saveProgress();

                this.renderStats();

                this.generatePuzzle();

            }

            // --- UI & Utilities ---

            renderStats() {

                this.dom.level.innerText = this.level;

                this.dom.hints.innerText = this.hints;

                this.dom.checkpoint.innerText = this.checkpoint;

                // Integrity Bar

                const pct = ((MAX_MISTAKES - this.mistakes) / MAX_MISTAKES) * 100;

                this.dom.integrityBar.style.width = `${pct}%`;

                this.dom.integrityText.innerText = `${Math.round(pct)}%`;

               

                // Color changes based on health

                if(pct > 60) this.dom.integrityBar.className = "h-full bg-emerald-500 transition-all duration-300";

                else if(pct > 30) this.dom.integrityBar.className = "h-full bg-yellow-500 transition-all duration-300";

                else this.dom.integrityBar.className = "h-full bg-red-600 transition-all duration-300";

                // Render X marks for mistakes

                this.dom.lives.innerHTML = '';

                for(let i=0; i<this.mistakes; i++) {

                    this.dom.lives.innerHTML += '<i class="fas fa-times-circle"></i>';

                }

            }

            showNotification(title, msg, type) {

                this.dom.overlayTitle.innerText = title;

                this.dom.overlayMsg.innerText = msg;

                this.dom.overlay.classList.remove('hidden');

                this.dom.overlay.classList.add('flex');

                if (type === "error") {

                    this.dom.overlayTitle.className = "text-3xl font-bold mb-2 text-red-500";

                    this.dom.overlayIcon.innerHTML = '<i class="fas fa-skull-crossbones text-red-600"></i>';

                    this.dom.overlayBtn.className = "bg-red-600 hover:bg-red-700 text-white px-8 py-3 rounded font-bold transition-colors";

                    this.dom.overlayBtn.innerText = "REBOOT SYSTEM";

                } else {

                    this.dom.overlayTitle.className = "text-3xl font-bold mb-2 text-cyan-400";

                    this.dom.overlayIcon.innerHTML = '<i class="fas fa-save text-cyan-400"></i>';

                    this.dom.overlayBtn.className = "bg-cyan-600 hover:bg-cyan-500 text-white px-8 py-3 rounded font-bold transition-colors";

                    this.dom.overlayBtn.innerText = "CONTINUE";

                }

            }

            saveProgress() {

                const data = {

                    level: this.level,

                    hints: this.hints,

                    checkpoint: this.checkpoint

                };

                localStorage.setItem('cerebrum_save', JSON.stringify(data));

            }

            loadProgress() {

                const saved = localStorage.getItem('cerebrum_save');

                if (saved) {

                    const data = JSON.parse(saved);

                    this.level = data.level;

                    this.hints = data.hints;

                    this.checkpoint = data.checkpoint;

                }

            }

        }

        // Initialize

        const game = new Game();

    </script>
